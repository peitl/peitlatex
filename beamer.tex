% typeset the footline to contain author names, title, and page numbers
%\renewcommand{\arraystretch}{1.5}
\makeatother
\setbeamertemplate{footline}
{
  \leavevmode%
  \hbox{%
  \begin{beamercolorbox}[wd=.4\paperwidth,ht=2.25ex,dp=1ex,center]{author in head/foot}%
    \usebeamerfont{author in head/foot}\insertshortauthor
  \end{beamercolorbox}%
  \begin{beamercolorbox}[wd=.6\paperwidth,ht=2.25ex,dp=1ex,center]{title in head/foot}%
    \usebeamerfont{title in head/foot}\insertshorttitle\hspace*{3em}%
	\insertframenumber{} / \inserttotalframenumber\hspace*{1ex}
  \end{beamercolorbox}}%
  \vskip0pt%
}

\newcommand{\titleframe}[1]{
	\begin{frame}
		\vfill
		\begin{center}
		~
		\begin{beamercolorbox}[wd=0.9\textwidth,sep=8pt,center,shadow=true,rounded=true]{sectiontitle}
			\usebeamerfont{title}#1\par%
		\end{beamercolorbox}
		~
		\end{center}
		\vfill
	\end{frame}
}

% add a title frame at the start of each section with the section name

}
